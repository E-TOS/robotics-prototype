262 void RobotMotor::checkForActualPress(void) {
263  if (sinceTrigger >= TRIGGER_DELAY) {
264  // if the last interrupt was a press (meaning it's stabilized and in contact)
265  // then there's a real press
266  if (triggerState != 0) {
267  actualPress = true;
268  limitSwitchState = triggerState;
269  }
270  // otherwise it's not a real press

272  // and so should actualPress

274  ;
275  }

277  triggered = false;
